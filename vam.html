/ RPG Game in JavaScript // Create player object with properties for name, health, and attack power var player = { name: "Player", health: 100, attackPower: 10 }; // Create enemy object with properties for name, health, and attack power var enemy = { name: "Enemy", health: 50, attackPower: 5 }; // Function for player's attack function playerAttack() { // Subtract enemy's health by player's attack power enemy.health -= player.attackPower; console.log(player.name + " attacked " + enemy.name + " and dealt " + player.attackPower + " damage!"); console.log(enemy.name + "'s health is now at " + enemy.health + "."); } // Function for enemy's attack function enemyAttack() { // Subtract player's health by enemy's attack power player.health -= enemy.attackPower; console.log(enemy.name + " attacked " + player.name + " and dealt " + enemy.attackPower + " damage!"); console.log(player.name + "'s health is now at " + player.health + "."); } // Game loop while (player.health > 0 && enemy.health > 0) { // Ask player if they want to attack or defend var action = prompt("Do you want to attack or defend?"); if (action === "attack") { playerAttack(); } else if (action === "defend") { console.log(player.name + " defended the attack!"); } else { console.log("Invalid action, please choose either attack or defend."); } // Check if enemy is still alive before attacking if (enemy.health > 0) { enemyAttack(); } // Check if player is still alive if (player.health <= 0) { console.log(player.name + " has been defeated!"); } else if (enemy.health <= 0) { console.log(enemy.name + " has been defeated!"); } }
